<div class="welcome-container" markdown="0">

  <h1 class="welcome-title">Cloudflare WARP VPN</h1>
  <p class="welcome-subtitle">Install and enable Cloudflare WARP to secure your internet connection.</p>

  <div class="welcome-section">
    <h2>Install from Play Store</h2>
    <div class="welcome-checklist">

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">1</span></div>
        <div>
          <strong>Open the Play Store</strong>
          <p>Open the <strong>Apps menu</strong> and select <strong>Play Store</strong>.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-01.png" image_path="/assets/images/chrome-vpn-01.png" %}

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">2</span></div>
        <div>
          <strong>Install Cloudflare One Agent</strong>
          <p>Search for <strong>Cloudflare One Agent</strong>, select it from the results, and click <strong>Install</strong>.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-02.png" image_path="/assets/images/chrome-vpn-02.png" %}
{% include figure url="/assets/images/chrome-vpn-03.png" image_path="/assets/images/chrome-vpn-03.png" %}

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">3</span></div>
        <div>
          <strong>Open the app</strong>
          <p>Once installation is complete, click <strong>Open</strong>.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-04.png" image_path="/assets/images/chrome-vpn-04.png" %}

    </div>
  </div>

  <div class="welcome-section">
    <h2>Configure VPN</h2>
    <div class="welcome-checklist">

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">4</span></div>
        <div>
          <strong>Install VPN profile</strong>
          <p>Select <strong>Install VPN Profile</strong>, then click <strong>OK</strong> when prompted.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-05.png" image_path="/assets/images/chrome-vpn-05.png" %}
{% include figure url="/assets/images/chrome-vpn-06.png" image_path="/assets/images/chrome-vpn-06.png" %}

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">5</span></div>
        <div>
          <strong>Allow notifications</strong>
          <p>When prompted, select <strong>Allow notifications</strong> and then <strong>Allow</strong>. This ensures you receive re-authentication alerts.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-07.png" image_path="/assets/images/chrome-vpn-07.png" %}
{% include figure url="/assets/images/chrome-vpn-08.png" image_path="/assets/images/chrome-vpn-08.png" %}

      <div class="checklist-item">
        <div class="checklist-icon"><span style="font-weight:700; font-size:1rem;">6</span></div>
        <div>
          <strong>Enable Cloudflare Zero Trust</strong>
          <p>Toggle on <strong>Cloudflare Zero Trust</strong> to activate the VPN.</p>
        </div>
      </div>

{% include figure url="/assets/images/chrome-vpn-09.png" image_path="/assets/images/chrome-vpn-09.png" %}

      <div class="checklist-item" style="background: #e8f5e9; border-color: #c8e6c9;">
        <div class="checklist-icon" style="background: #fff; border-color: #c8e6c9; color: #00a828;"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="20 6 9 17 4 12"/></svg></div>
        <div>
          <strong>You're connected!</strong>
          <p>Cloudflare WARP is now active and securing your connection.</p>
        </div>
      </div>

    </div>
  </div>
